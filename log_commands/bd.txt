CREATE DATABASE bd_hospital;
CREATE TABLE usuario ( id int NOT NULL AUTO_INCREMENT, matricula int NOT NULL, senha int NOT NULL, PRIMARY KEY (id) )default charset = utf8
CREATE TABLE medico (
    ID_Med int NOT NULL AUTO_INCREMENT,
    CRM varchar(30) NOT NULL,
    FOREIGN KEY (ID_Med) REFERENCES usuario(ID_Usuario),
    CONSTRAINT ID_Med UNIQUE (ID_Med)
    );

CREATE TABLE enfermeiro ( ID_Enf int NOT NULL AUTO_INCREMENT, COREN varchar(30) NOT NULL, FOREIGN KEY (ID_Enf) REFERENCES usuario(ID_Usuario), CONSTRAINT ID_Enf UNIQUE (ID_Enf) )

select * from usuario where id_usuario not in (select ID_Med from medico )     

CREATE TABLE paciente (
    num int NOT NULL AUTO_INCREMENT,
    data_abertira date,
    nome varchar(30) NOT NULL,
    nascimento date,
    sexo enum('M','F'),
	endereco varchar(200),
    telefone varchar(200),
    email varchar(200),
    PRIMARY KEY (num)
    );
